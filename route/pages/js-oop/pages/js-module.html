Soal

Buat sebuah modul dengan ketentuan:
Karyawan.js berisikan class Karyawan yang memiliki beberapa properti dan method
Export class tersebut menggunakan export default
Import pada index.js kemudian buat object dari class yang sudah dibuat

Jawaban

                    
                        // export karyawan.js
                        class Karyawan {
                          constructor(name, role, salary) {
                            this.name = name;
                            this.role = role;
                            this.salary = salary;
                          }
                        
                        
                          introduce() {
                            return `Hi! I'm ${this.name} as a ${this.role} and paid for ${this.salary} dollars`;
                          }
                        }

                        // import karyawan.js
                        import {karyawan, introduce} from "./karyawan.js";

                        let bill = new Karyawan("Bill Gates", "IT Helpdesk", 1000);
                        let elon = new Karyawan("Elon Musk", "Professional Meme Developer", 3000);
                        let takumi = new Karyawan("Takumi Fujiwara", "Professional courier", 5000);
